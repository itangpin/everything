 51     Node.prototype.getPath = function(){
 52         this.path = [];
 53         var node = this;
 54         while(node){
 55             if(node.parent){
 56                 this.path.unshift(node.parent);
 57             }
 58             node = node.parent;
 59         }
 60         return this.path;

143     /**
144      * Handle events
145      * @param {event object} event event object from browser
146      */
147     Node.prototype.onEvent = function(event){
148         var type = event.type;
149         var target = event.target || event.srcElement;
150 
151         // Keyboard events
152         if(type == 'keydown'){
153             var keyNum = event.keyCode;
154             // Ctrl + Enter
155             if(keyNum==13 && event.ctrlKey){
156                 event.preventDefault();
157                 this.createSiblingNodeAfter();
158                 return false;
159             }
160             // Shift + Enter
161             if(keyNum==13 && event.shiftKey){
162                 event.preventDefault();
163                 this._onInsertBefore({});
164                 return false;
165             }
166             // Tab
167             if(9 == event.keyCode){
168                 event.preventDefault();
169                 // Shift + Tab
170                 if(event.shiftKey){
171                     this.unindent();
172                 }else{
173                     this.indent();
174                 }
175                 return false;
176             }
177             // Delete
178             if(46 == event.keyCode){
179                 event.preventDefault();
180                 this.parent.removeChild(this);
181                 return false;
182             }
183             // Backspace on an 'empty' node
184             if(8 == event.keyCode){
185                 if(this.getContent() == ""){
186                     this.parent.removeChild(this);
187                 }
188             }
189             // move events
190             if(this.app.mode == "move"){
191             }
192         }
193 
194         // Mouse click events
195         if(type == 'click'){
196             if(target == this.buttonElement
197                 || target == this.buttonElement.childNodes[0]
198                 || target == this.buttonElement.childNodes[0].childNodes[0]){
199                 if(!event.altKey){
200                     this._onZoomIn();
201                 }else{
202                     this.collapse();
203                 }
204 
205             }
206             if(target == this.contentElement){
207                 this.app.setCurentNode(this);
208             }
209         }
210 
211 
212 
213         // send events to packages' handler
214         this.packageEventsHandle(event);
215     };

272     Node.prototype.setValue = function(value){
273         if($.isArray(value)){
274             // root node's value is an array
275             value = {
276                 content: "",
277                 children: value
278             };
279             // value type is used for get value
280             this.valueType = 'array';
281             this.value = value;
282             this.setValue(this.value);
283             return;
284         }
285         if($.type(value) == "string"){
286             this.valueType = 'string';
287             this.setContent(value);
288             return;
289         }
290 
291         if(value.content!=undefined){
292             this.setContent(value.content || "");
293         }
294         if($.isArray(value.children)){
295             if(value.children){
296                 this.setChildren(value.children);
297             }
298         }

394     /**
395      * Add a Node as a sibling Node right after this node
396      * @param {Node} node node to moved
397      */
398     Node.prototype.addSiblingNodeBefore = function(node){
399         if(this.isRootNode){
400             // you can't add a sibling node for rootNode
401             // that's why we call it a ROOT node
402             return;
403         }
404         if(node.isRootNode){
405             // it's not a rootNode anymore after you move it
406             node.isRootNode = false;
407         }
408         if(node.parent){
409             node.parent.removeChild(node);
410         }
411         node.setParent(this.parent);
412         this.parent._addChild(node);
413         // TODO
414         // add updateDom method to Node
415         $(this.row).before(node.row);

484     /**

488     Node.prototype.removeChild = function(node){
489         var childNode = this.childrenMap[node.id];
490         this.childs = this.childs.filter(function(child){
491             return child.id != node.id;
492         });
493         childNode.row.parentNode.removeChild(childNode.row);
494         this.childrenMap[node.id] = undefined;
495         this._updateDomIndexes();

661            });
662         }else{
663            this.childrenHeight = $(this.childrenElement).height();
664            $(self.row).addClass('collapse');
665            $(this.childrenElement).animate({height:0},200,function(){
666                $(this).addClass('collapse');
667                self.collapseAnimating = false;
