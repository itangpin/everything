 33     Node.prototype._create = function(position){
 34         var self = this;
 35         this._createDom();
 36         //this._bindEvent();
 37 
 38         // add the created Node to the DOM
 39         if(position){
 40             switch (position.type){
 41                 case 'after':
 42                     insertAfter(position.el);
 43                     break;
 44                 case 'append':
 45                     appendInside(position.el);
 46                     break;
 47                 default:
 48                     break;
 49             }
 50         }else{
 51             this.parent.appendChild(this.row);
 52         }
 53 
 54         function insertAfter(el){
 55             $(el).after(self.row);
 56         }
 57         function appendInside(el){
 58             $(el).append(self.row);
 59         }
 60     };

103     /**
104      * Handle events
105      * @param {event object} event event object from browser
106      */
107     Node.prototype.onEvent = function(event){
108         var type = event.type;
109         var target = event.target || event.srcElement;
110 
111         // Keyboard events
112         if(type == 'keydown'){
113             var keyNum = event.keyCode;
114             // Enter
115             if(keyNum==3){
116                 event.preventDefault();
117                 // create new line
118                 this.createSiblingNodeAfter();
119                 return false;
120             }
121             // Tab
122             if(9 == e.keyCode){
123                 e.preventDefault();
124                 // Shift + Tab
125                 if(e.shiftKey){
126                     this.unindent();
127                 }else{
128                     this.indent();
129                 }
130                 return false;
131             }
132             // Delete
133             if(46 == e.keyCode){
134                 e.preventDefault();
135                 self.parent.removeChild(self.id);
136                 return false;
137             }
138             // Backspace on an 'empty' node
139             if(8 == e.keyCode){
140                 if(self.getContent() == ""){
141                     self.parent.removeChild(self.id);
142                 }
143             }
144         }
145 
146         // Mouse click events
147         if(type == 'click'){
148             if(target == this.buttonElement){
149                 this.collapse();
150             }
151         }
152 
153     };

202     /**
203      * Set content and children of the node,
204      * @param {String|Array|Object} value content of the Node

206     Node.prototype.setValue = function(value){
207         if(_.isArray(value)){
208             value = {
209                 content: "",
210                 children: value
211             };
212             this.data = value;
213             this.setValue(value);
214             return;
215         }
216         if(_.isString(value)){
217             this.setContent(value);
218             return;
219         }
220         if(value.content!=undefined && _.isArray(value.children)){
221             this.setContent(value.content || "");
222             if(value.children){
223                 this.setChildren(value.children);
224             }
225         }
226 
227     };
